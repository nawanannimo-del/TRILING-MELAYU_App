} // --- ⚙️ CONFIG & DATA --- const wordBank = [ { eng: "Apple", thai: "แอปเปิ้ล", rumi: "Epal", jawi: "ايفل", emoji: "🍎" }, { eng: "Cat", thai: "แมว", rumi: "Kucing", jawi: "کوچيڠ", emoji: "🐱" }, { eng: "School", thai: "โรงเรียน", rumi: "Sekolah", jawi: "سکوله", emoji: "🏫" }, { eng: "Water", thai: "น้ำ", rumi: "Air", jawi: "اءير", emoji: "💧" } ]; let appState = { questions: [], currentIndex: 0, score: 0, isProcessing: false // ป้องกันการรัวปุ่มจนแอปพัง }; // --- 🌪️ SHUFFLE ENGINE --- function shuffle(array) { return array.sort(() => Math.random() - 0.5); } // --- 🎯 QUIZ LOGIC --- function startQuiz() { appState.score = 0; appState.currentIndex = 0; // สุ่มคำศัพท์จากคลัง appState.questions = shuffle([...wordBank]); renderQuestion(); } function renderQuestion() { const container = document.getElementById('quiz-content'); // ID ตามโครงเดิมของน้อง const optionsGrid = document.getElementById('quiz-options'); const currentWord = appState.questions[appState.currentIndex]; // 🎲 RANDOMIZE QUESTION TYPE (สลับรูปแบบทุกข้อ!) const type = Math.floor(Math.random() * 3); let qText, correctAns, langLabel; if(type === 0) { qText = currentWord.eng; correctAns = currentWord.thai; langLabel = "English ➡️ Thai"; } else if(type === 1) { qText = currentWord.jawi; correctAns = currentWord.eng; langLabel = "Jawi ➡️ English"; } else { qText = currentWord.thai; correctAns = currentWord.rumi; langLabel = "Thai ➡️ Rumi"; } // เคลียร์ค่าเก่าและใส่แอนิเมชัน container.innerHTML = `
${currentWord.emoji}

${langLabel}

${qText}

`; // สร้างตัวเลือก (สุ่มตัวหลอกไม่ให้ซ้ำกับคำตอบจริง) let options = [correctAns]; while(options.length < 4) { let randomWord = wordBank[Math.floor(Math.random() * wordBank.length)]; let wrongAns = (type === 0) ? randomWord.thai : (type === 1 ? randomWord.eng : randomWord.rumi); if(!options.includes(wrongAns)) options.push(wrongAns); } options = shuffle(options); optionsGrid.innerHTML = ''; options.forEach(opt => { const btn = document.createElement('button'); btn.className = 'quiz-option'; btn.innerText = opt; btn.onclick = () => checkAnswer(opt, correctAns, btn); optionsGrid.appendChild(btn); }); } function checkAnswer(selected, correct, btn) { if(appState.isProcessing) return; // ล็อกปุ่มชั่วคราว appState.isProcessing = true; if(selected === correct) { btn.classList.add('correct', 'correct-glow'); appState.score += 10; playFeedback('correct'); // ถ้ามีไฟล์เสียง } else { btn.classList.add('wrong', 'shake'); playFeedback('wrong'); } // หน่วงเวลาเล็กน้อยเพื่อให้ดูผลลัพธ์ ก่อนสไลด์ไปข้อถัดไป setTimeout(() => { appState.currentIndex++; appState.isProcessing = false; if(appState.currentIndex < appState.questions.length) { renderQuestion(); } else { showSummary(); } }, 1000); } // ระบบจัดการหน้าจอ (เสถียรและเร็ว) function showScreen(screenId) { document.querySelectorAll('.screen').forEach(s => { s.style.display = 'none'; s.classList.remove('active'); }); const target = document.getElementById(screenId); target.style.display = 'block'; setTimeout(() => target.classList.add('active'), 10); }
